13 #define Blend_Normal(A,B) ((uint8)(A))
14 #define Blend_Lighten(A,B) ((uint8)((B > A) ? B:A))
15 #define Blend_Darken(A,B) ((uint8)((B > A) ? A:B))
16 #define Blend_Multiply(A,B) ((uint8)((A * B) / 255))
17 #define Blend_Average(A,B) ((uint8)((A + B) / 2))
18 #define Blend_Add(A,B) ((uint8)(FMath::Min(255, (A + B))))
19 #define Blend_Subtract(A,B) ((uint8)((A + B < 255) ? 0:(A + B - 255)))
20 #define Blend_Difference(A,B) ((uint8)(FMath::Abs(A - B)))
21 #define Blend_Negation(A,B) ((uint8)(255 - FMath::Abs(255 - A - B)))
22 #define Blend_Screen(A,B) ((uint8)(255 - (((255 - A) * (255 - B)) >> 8)))
23 #define Blend_Exclusion(A,B) ((uint8)(A + B - 2 * A * B / 255))
24 #define Blend_Overlay(A,B) ((uint8)((B < 128) ? (2 * A * B / 255):(255 - 2 * (255 - A) * (255 - B) / 255)))
25 #define Blend_SoftLight(A,B) ((uint8)((B < 128)?(2*((A>>1)+64))*((float)B/255):(255-(2*(255-((A>>1)+64))*(float)(255-B)/255))))
26 #define Blend_HardLight(A,B) (Blend_Overlay(B,A))
27 #define Blend_ColorDodge(A,B) ((uint8)((B == 255) ? B:FMath::Max(255, ((A << 8 ) / (255 - B)))))
28 #define Blend_ColorBurn(A,B) ((uint8)((B == 0) ? B:FMath::Max(0, (255 - ((255 - A) << 8 ) / B))))
29 #define Blend_LinearDodge(A,B)(Blend_Add(A,B))
30 #define Blend_LinearBurn(A,B) (Blend_Subtract(A,B))
31 #define Blend_LinearLight(A,B)((uint8)(B < 128)?Blend_LinearBurn(A,(2 * B)):Blend_LinearDodge(A,(2 * (B - 128))))
32 #define Blend_VividLight(A,B) ((uint8)(B < 128)?Blend_ColorBurn(A,(2 * B)):Blend_ColorDodge(A,(2 * (B - 128))))
33 #define Blend_PinLight(A,B) ((uint8)(B < 128)?Blend_Darken(A,(2 * B)):Blend_Lighten(A,(2 * (B - 128))))
34 #define Blend_HardMix(A,B) ((uint8)((Blend_VividLight(A,B) < 128) ? 0:255))
35 #define Blend_Reflect(A,B) ((uint8)((B == 255) ? B:FMath::Min(255, (A * A / (255 - B)))))
36 #define Blend_Glow(A,B) (Blend_Reflect(B,A))
37 #define Blend_Phoenix(A,B) ((uint8)(FMath::Min(A,B) - FMath::Max(A,B) + 255))
